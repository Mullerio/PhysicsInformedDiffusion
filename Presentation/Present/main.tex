\begin{frame}{Physics Informed Diffusion Models}
    \begin{definition}
        We assume our physics is described by a PDE on $\Omega \subseteq \mathbb{R}^d$ of the form 
        \[
            \mathcal{F}\left(u(\xi)\right) = 0 \quad \text{with} \quad \xi \in \Omega. 
        \]
        and we have a boundary operator 
        \[\mathcal{B}\left(u(\xi)\right) = 0 \quad \text{with} \quad \xi \in \partial \Omega.\]
        where $u : \Omega \rightarrow \mathbb{R}^c$ is the solution of interest. To simplify, we write \[R(u(\xi)) = [\mathcal{F}(u(\xi)), \mathcal{B}(u(\xi))]^\top\].
    \end{definition}
    They only consider $$[1,...,n] \times [1,...,n] \subseteq \mathbb{Z}_n \times \mathbb{Z}_n.$$
    Similar to standard image based diffusion models, data is then considered to be given as $x_0 \in \mathbb{R}^{c \times n \times n}$ and $\mathcal{F}, \mathcal{B}$ are discretized accordingly.
\end{frame}



\begin{frame}{Setting up Physics based Training Objective}
    The residual $\mathcal{R} = 0$ in the non discrete gives us an infinite amount of information. 
    To utilize this, we introduce a loss based on $\mathcal{R}$ via a \textit{virtual observable}.

    $$q_{\mathcal{R}}(\cdot \mid x_0) \sim \mathcal{N}(\mathcal{R}(x_0), \sigma^2 I_d)$$


    \begin{definition}
        Define $$p_\theta(\hat{r}) = \int_{\Omega} q_{\mathcal{R}}(\hat{r} \mid x_0) p_\theta(x_0) dx_0$$
        then we can optimize 
        $$ \arg \max_\theta \mathbb{E}_{\hat{r} \sim \delta_0}\left[\log p_\theta(\hat{r})\right]=\arg \max _\theta \mathbb{E}_{x_0 \sim p_\theta}\left[\log q_{\mathcal{R}}\left(\hat{r}=0 \mid x_0\right)\right] .$$
    \end{definition}
    So we can train our model by evaluating $x_0$ on the the virtual observable log likelihood.
    Note that this loss is equivalent to $$\arg \max_\theta \log p_\theta(0) = \arg \max _\theta \mathbb{E}_{x_0 \sim p_\theta}\left[\log q_{\mathcal{R}}\left(0 \mid x_0\right)\right]$$
\end{frame}

\begin{frame}{Introducing the Data Loss}
    We explicitly consider that we have a dataset, which we also want to consider in our training objective. 
    For this, we just additionally introduce the standard model log likelihood as an additional term. 
    \begin{definition}
        Our objective is given by
        $$\arg \max _\theta \mathbb{E}_{x_0 \sim p_\theta} \left[\log p_\theta(x_0)\right] +\arg \max _\theta \mathbb{E}_{x_0 \sim p_\theta}\left[\log q_{\mathcal{R}}\left(0 \mid x_0\right)\right]$$
    \end{definition}

\end{frame}
\begin{frame}{Sample and Mean Estimation}
    Classical diffusion models try and approximate $$\hat{x}_0 \approx \mathbb{E}[x_0 \mid x_t]$$
    where $\hat{x}_0$ is the model. Then, use $\hat{x}_0$ directly for the loss. This is called \text{mean estimation}. However, if we just compute our physics loss via plugging the model output into the residual we have  $$\mathcal{R}(\hat{x}_0) \approx \mathcal{R}(\mathbb{E}[x_0 \mid x_t]) \neq \mathbb{E}[\mathcal{R}(x_0 \mid x_t)]$$
    which would be the correct expected residual. To mitigate this, they increase the variance as $t$ increases.  
    Another solution, which is more costly during training, is to use DDIM sampling, which will yield us a sample $x_0^*$ of $p_\theta$, which yields us a better estimate for the residual, this they call sample estimation.    
\end{frame}

\begin{frame}{Full Training Algorithm}

\begin{algorithm}[H]
\caption{Physics-informed diffusion model training}
\label{alg:pidm}
\begin{algorithmic}[1]
\Repeat
    \State Sample $x_0 \sim q(x_0), t \sim \text{Uniform}\{1, \ldots, T\}, \epsilon \sim \mathcal{N}(0, I)$
    \State Compute
    \[
        x_t = \sqrt{\bar{\alpha}_t}\, x_0 + \sqrt{1 - \bar{\alpha}_t}\, \epsilon
    \]
    \State Estimate $x_0^*$ via
        $\hat{x}_0 \approx \mathbb{E}[x_0 \mid x_t]$
    or DDIM sampling. 
    \State Take a gradient descent step on
    \[
        \nabla_\theta \Bigg[
        \lambda_t \left\| x_0 - \hat{x}_0(x_t, t) \right\|^2
        + c \cdot \frac{1}{2\Sigma_t}
        \left\| \mathcal{R}\!\left(x_0^*(x_t, t)\right) \right\|^2
        \Bigg]
    \]
\Until{converged or for fixed number of iterations}
\end{algorithmic}
\end{algorithm}
Where $c > 0$ is some hyperparameter which controls the weight of the physics loss.

\end{frame}

\begin{frame}{Experimental Setup}
    They consider only one set of PDEs, namely the Darcy Flow equations given by 
    \begin{align*}
        u(x) = - K(x) \nabla p(x) \\
        -\nabla \cdot u(x) = f(x) \\
    \end{align*}
    where $K$ is the permeability field, $p$ the pressure field and $f$ a source term. They describe fluid flow through porus media.

\begin{frame}{Discussion}
     \begin{itemize}
        \item They only consider 2d data and PDEs on $\mathbb{R}^2$, how this works on higher dimensions is not tested. 
        \item The hyperparameter $c$ for the variance is crucial, choosing it too large leads to an overfocus on the residual and thus not sampling from the entire data distribution.
    \end{itemize}
